//======================  generator.cpp  ======================
#include<random>
#include<iostream>
#include<ctime>
using namespace std;
int rand()
{
    int min = 0, max = 1e9;
    static std::mt19937_64 rng(time(0));
    return std::uniform_int_distribution<int>(min, max)(rng);
}
const int MAXN = 10,MAXA = 200;
int main(){
    cout<<1<<endl;//多组测试用例
    srand(time(0));
    int n = rand() % MAXN+1;
    cout<<n<<endl;
    for(int i=1;i<=n;i++){
        cout << rand() % MAXA <<" \n"[i==n];
    }
}
