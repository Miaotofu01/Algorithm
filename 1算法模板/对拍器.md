//======================  generator.cpp  ======================
#include<random>
#include<iostream>
#include<ctime>
#include<algorithm>
#include<numeric>
#include <vector>
using namespace std;
static mt19937_64 rng(time(0));
int ra(int min=0,int max=1e9){//随机数生成器
    return std::uniform_int_distribution<int>(min, max)(rng);
}
const int MAXN = 10,MAXA = 200;
void gettree(){//生成随机树结构
    int n = ra(1, MAXN);
    cout << n << endl;
    vector<int>perm(n+1);
    iota(perm.begin(), perm.end(), 0);
    shuffle(perm.begin() + 1, perm.end(), rng);
    vector<int> g(n + 1);
    for(int i=2;i<=n;i++)
        g[i] = ra(1, i-1);
    for(int i=2;i<=n;i++)
        cout << perm[i] << " " << perm[g[i]] << endl;
}
void getarray(){//生成数组 -- q组区间查询
    int n = ra(1, MAXN);
    cout << n << endl;
    for (int i = 1; i <= n; i++)
    {
        cout << ra(1, MAXA) << " \n"[i == n];
    }
    int q = ra(1, n);
    cout << q << endl;
    while (q--)
    {
        int l = ra(1, n);
        int r = ra(l, n);
        cout << l << " " << r << endl;
    }
}
int main(){
    cout<<1<<endl;//多组测试用例
    gettree();
}
